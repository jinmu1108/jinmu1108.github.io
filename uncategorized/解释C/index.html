解释C

#include <stdio.h>

int main() {
int x;
x = 12345;
int digit;
int ret = 0;/局部变量定义/

while (x > 0) { /*条件*/
    digit = x % 10;/*赋给digit,x的个位数*/
    ret = ret * 10 + digit;
    printf("x=%d,digit=%d,ret=%d\n,x,digit,ret");
    x /= 10/*去掉x的个位数*/;
}
printf("%d", ret);

return 0;

}

while循环

#include <stdio.h>

int main() {
int number;
int sum = 0;
int count = 0;/局部变量定义/

printf("请输入number的值：");
scanf("%d", &number);/*初始化number*/
while (number != -1) { /*条件*/
    sum += number;
    count ++;
    scanf("%d", &number)/*改变判断条件*/;
}

printf("%f\n", 1.0 * sum / count);/*求平均数*/

return 0;

}

逆序整数

#include <stdio.h>

int main() {
int x;
scanf(“%d”, &x);

int mask = 1;
int t = x;/*将x赋值给t防止x变为0*/
while (t > 9) {
    t /= 10;
    mask *= 10;
}/*将mask变为与x相同位数*/
printf("x=%d,mask=%d\n", x, mask);
do {
    int d = x / mask;/*取x最高位*/
    printf("%d", d);
    if (mask > 9) {
        printf(" ");
    }
    x %= mask;/*去掉x的最高位*/
    mask /= 10;/*去掉一个0*/
} while (mask > 0);
printf("\n");

return 0;

}

int a, b;
int min;

scanf("%d %d", &a, &b);
if (a < b) {
    min = a; 
} else {
    min = b;
}
int ret = 0;
int i;
for (i = 1; i < min; i++) {
    if (a % i == 0) {
        if (b % i == 0) {
            ret = i;
        }
    }
}

printf("%d和%d的最大公约数是：%d\n", a, b, ret);
return 0;

}

最小公倍数

#include <stdio.h>

int main() {
int n, x, y, z, sum, xy, xz, yz, xyz;
while (scanf(“%d”, &n) != EOF) {
sum = 0;
x = n / 5;
y = n / 6;
z = n / 8; /求出被5，6，8整除的数有几个/
xy = n / 30;
xz = n / 40;
yz = n / 24; /分别两个数之间除以最小公倍数求出重复的数有几个/
xyz = n / 120; /3个数的最小公倍数，求出前三步多减的有几个数/
sum = x + y + z - xy - xz - yz + xyz;

    printf("%d\n", sum);
}

数组

#include <stdio.h>

int main() {
int x;
double sum = 0;
int cnt = 0 ;
int number[100];//定义数组有100个int
scanf(“%d”, &x);
while (x != -1) {
number[cnt] = x;//对数组中的元素赋值，让第cnt位置上的元素等于x
sum += x;
cnt++;
scanf(“%d”, &x);
}
if (cnt > 0) {
printf(“%f\n”, sum / cnt);
int i;
for (i = 0; i < cnt; i++) {
if (number[i] > sum / cnt) {
printf(“%d\n”, number[i]);/使用数组中的元素，要拿number中的第i个值与sum/cnt进行比较，输出第i个元素/
}
}
}
return 0;
}

利用数组找到素数

#include <stdio.h>

int isPrime(int x, int knownPrime[], int number0fKnows) {
int ret = 1;
int i;
for (i = 0; i < number0fKnows; i++) {
if (x % knownPrime[i] == 0) {
ret = 0;
break;
}
}
return ret;

}/定义引用函数isPrime/
int main() {
const int number = 10;/定义number为常量10/
int prime[number] = {2}; /定义prime中0位变量为2/
int count = 1; /定义count为1/
int i = 3;/定义i等于三/

while (count < number) { /*条件*/
    if (isPrime(i, prime, count)) { /*如果isPrime函数给我们一个素数我们就把它写进Prime里测试下一个i*/
        prime[count++] = i;/*将count这一位的prime数组中的变量赋值为i*，并且副作用是让count++*//
    }
    i++;
}
for (i = 0; i < number; i++) {
    printf("%d", prime[i]);
    if ((i + 1) % 5)
        printf("\t");
    else
        printf("\t");
}/*在输出一个素数后空格使其看着更清爽*/
return 0;

}
